CREATE DATABASE if not exists instagram;
USE instagram;

CREATE TABLE user(
	id INT,
    age INT,
    name VARCHAR(30) NOT NULL,
    email varchar(50) UNIQUE,
    followers INT DEFAULT 0,
    following INT,
    CONSTRAINT CHECK (age>=13),
    PRIMARY KEY (id)
);

CREATE TABLE post(
	id INT PRIMARY KEY,
    content VARCHAR(100),
    user_id INT,
    FOREIGN KEY(user_id) REFERENCES user(id)
);
