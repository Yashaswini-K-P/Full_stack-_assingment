Qs. Create a function to find the min number in an array.
Ans.
function getMin(arr){
    let res=arr.reduce((min, el)=>{
    if(min>el){
        return el; 
    }else{
        return min;
    }
    });
    return res;
}
let arr=[5, 8, 6, 9, 30, 64, 71, 6, 5];
console.log(getMin(arr));
