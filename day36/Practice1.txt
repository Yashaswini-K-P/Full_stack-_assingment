Qs. Create a database for your college.
    Create a table named Teacher to store (id, name, subject, salary)
    Insert following data in the table:
    23, "ajay", "math", 50000
    47, "bharat", "english", 60000
    18, "chetan", "chemistry", 45000
    9, "divya", "physics", 75000
Sol.
    CREATE DATABASE IF NOT EXISTS College;
    USE College;
    CREATE TABLE Teacher(
        id INT PRIMARY KEY,
        name VARCHAR(30),
        subject VARCHAR(30),
        salary INT
    );

    INSERT INTO Teacher
    (id, name, subject, salary) 
    VALUES
    (23, "ajay", "math", 50000),
    (47, "bharath", "english", 60000 ),
    (18, "chetan", "chemistry", 45000),
    (9, "divya", "physics", 75000);

    SELECT * FROM Teacher;



Qs. Select teachers whose salary is more than 55k
Sol.
    SELECT * FROM Teacher 
    WHERE salary>55000;


Qs. Rename the salary column of teacher table to ctc
Sol.
    ALTER TABLE Teacher
    CHANGE COLUMN salary ctc INT;


Qs. Update salary of all teachers by giving them an increment of 25%
Sol.
    UPDATE Teacher
    SET ctc = ctc+0.25*ctc;
    SET SQL_SAFE_UPDATES=0;

Qs. Add a new column for teachers called city. The default city should be "Bengaluru"
Sol.
    ALTER TABLE Teacher
    ADD COLUMN city VARCHAR(30) DEFAULT "Bengaluru";

Qs. Delete the salary column for teacher table
Sol.
    ALTER TABLE Teacher
    DROP COLUMN ctc;
